108  template<typename T>
109  void extractDualCircle(c3ga::Mvec<T> dualcircle, T &radius, c3ga::Mvec<T> &center, c3ga::Mvec<T> &direction){
110 
111  c3ga::Mvec<T> pt1, pt2;
112  extractPairPoint(dualcircle, pt1, pt2);
113  radius = 0.5 * sqrt( (pt2[c3ga::E1]-pt1[c3ga::E1])*(pt2[c3ga::E1]-pt1[c3ga::E1]) + (pt2[c3ga::E2]-pt1[c3ga::E2])*(pt2[c3ga::E2]-pt1[c3ga::E2]) + (pt2[c3ga::E3]-pt1[c3ga::E3])*(pt2[c3ga::E3]-pt1[c3ga::E3]) );
114  center = point(0.5*(pt2[c3ga::E1] + pt1[c3ga::E1]), 0.5*(pt2[c3ga::E2] + pt1[c3ga::E2]), 0.5*(pt2[c3ga::E3] + pt1[c3ga::E3]));
115  direction = pt2 - pt1;
116  direction[c3ga::Ei] = 0;
117  direction /= direction.norm();
118  }

125  void extractPairPoint(const c3ga::Mvec<T> &pairPoint, c3ga::Mvec<T> &pt1, c3ga::Mvec<T> &pt2){
126  c3ga::Mvec<T> denominator = - c3ga::ei<double>() | pairPoint;
127  pt1 = (pairPoint + sqrt(fabs(pairPoint | pairPoint)) ) / denominator;
128  pt2 = (pairPoint - sqrt(fabs(pairPoint | pairPoint)) ) / denominator;

131  pt1.roundZero();

563
